\textbf{Convergence in Distribution}$. 30 March 2008.

\begin{asmp}\label{asmp_array}
For each model $j$, the random array 
$\{(\epjarray{s}, \xjarray{s}); j = 0, 1; s = 1, 2, \dotsc\}$ 
is stationary and absolutely regular with coefficients $\beta_\tau$ of
size $\rho/(\rho-2)$; $\rho$ is greater than two.
\end{asmp}

\begin{asmp}\label{asmp_loss}
There is a finite constant $B_L$ such that $\|\fxjarray{s}\|_\rho \leq B_L$
for all $j$, $s$, and $T$.
\end{asmp}

\begin{thm} \label{thm_normality}
If Assumptions \ref{asmp_array}-\ref{asmp_loss} hold and
$\lambda_{min}(\vc)^{-1} = O_p(1)$ with
    \[
    \vc \equiv P_T^{-1} \sum_{s,t = T + 1}^{T+P_T} 
    \me([\fxarray{s} - \marray{R_T}][\fxarray{t} - \marray{R_T}]' 
    \mid \barray{R_T})
    \]
    then $\vc^{-1/2} P_T^{1/2} (\elbararray - \marray{R}) 
    \xrightarrow{d} N(0,\I{2})$ as $P_T \to \infty$.
\end{thm}
